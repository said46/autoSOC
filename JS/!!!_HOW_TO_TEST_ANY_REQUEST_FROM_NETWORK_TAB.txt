// How to use:
// Go to DevTools → Network tab
// Find any API call
// Right-click → Copy → Copy as fetch
// Paste in console + remove ';' + add .then(r => r.json()).then(console.log)
// BAM! Instant test

// 1. Copy from DevTools
// 2. Paste in console
// 3. Add .then chain to see results

fetch("http://eptw.sakhalinenergy.ru/data/GetInstallationRoles?installationId=42&_=1761021296040", {
  "headers": {
    "accept": "*/*",
    "accept-language": "en-US,en;q=0.9,ru;q=0.8,zh-CN;q=0.7,zh;q=0.6",
    "x-requested-with": "XMLHttpRequest"
  },
  "referrer": "http://eptw.sakhalinenergy.ru/User/ChangeCurrentInstallationRole",
  "referrerPolicy": "strict-origin-when-cross-origin",
  "body": null,
  "method": "GET",
  "mode": "cors",
  "credentials": "include"
}).then(r => r.json()).then(console.log)
